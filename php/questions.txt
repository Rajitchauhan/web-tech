def findYValue(bits, maxSet, x):
    possible_y = [format(i , f'0{bits}b') for i in range(2**bits) if bin(i).count("1") <= maxSet]
    max_res = 0
    res_y=""
    for y in possible_y:
        xor_res = int(x , 2)*int(y , 2) # i use 8 insdead of ^
        if xor_res > max_res:
            max_res = xor_res
